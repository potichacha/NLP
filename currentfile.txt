    constructor() {
      mealzConsole('Setup start');
      window.miam.supplier.setupWithToken('eyAic3VwcGxpZXJJZCI6IDE5LCAib3JpZ2luIjogIm1hcm1pdG9uIiwgImRvbWFpbiI6ICJtaWFtLm1hcm1pdG9uLndlYiIsICJub1N1cHBsaWVyIjogdHJ1ZSB9');

      //window.miam.pos.load(26173);
      window.miam.basket.initialize();
      this.userLocation = null;

      window.miam.hook.setHookCallback(() => true);
      //window.miam.basket.reset()
      window.miam.features.collapseUnavailableProductsByDefault();
      this.initCTA();
      this.initPricing();
      this.initBasket();
      this.storeSubscribe();

      var that = this;
      window.didomiOnReady = window.didomiOnReady || [];
      window.didomiOnReady.push(function (Didomi) {
        mealzConsole('Mealz Didomi status complete', Didomi);
        mealzConsole('Mealz Didomi status', Didomi.getUserConsentStatusForPurpose('geolocation_data'));
        if (Didomi.getUserConsentStatusForPurpose('geolocation_data') === true) {
          that.initLocationUser();
        }
        Didomi.on('consent.changed', function() {
          if (Didomi.getUserConsentStatusForPurpose('geolocation_data') === true) {
            that.initLocationUser();
          }
        });
      });
    }

    initPricing() {
      mealzConsole('Init Pricing');
      if (window.mrtnRecipe === undefined) {
        mealzConsole('Recipe not defined for init pricing');
        return;
      }

      const pricing = document.querySelector("webc-miam-recipe-pricing");
      if (pricing === null || pricing === undefined) {
        mealzConsole('Not found Pricing');
      }

      pricing.recipeExtId = window.mrtnRecipe.id;
      pricing.serves = 4;
    }

    initCTA() {
      mealzConsole('Init CTA');
      if (window.mrtnRecipe === undefined) {
        mealzConsole('Recipe not defined');
        return;
      }
      mealzConsole(window.mrtnRecipe);
      const cardsCta = document.querySelectorAll("webc-miam-recipe-card-cta");

      if (cardsCta === null) {
        mealzConsole('Not found CTA');
        return;
      }

      cardsCta.forEach((cardCta) => {
        cardCta.recipeExtId = window.mrtnRecipe.id.toString();
        cardCta.serves = "4";
      });

      const inputIngredient = document.querySelector(".recipe-ingredients__qt-counter__value");
      if (inputIngredient) {
        cardsCta.forEach((cardCta) => {
          cardCta.serves = inputIngredient.value.toString();
        });
        const pricing = document.querySelector("webc-miam-recipe-pricing");
        if (pricing === null || pricing === undefined) {
          mealzConsole('Not found Pricing');
        }
        pricing.serves = inputIngredient.value.toString();
      }

      document.addEventListener('updateCounterIngredients', (event) => {
        this.updateCounterCTA(event);
      })
    }

    updateCounterCTA(event) {
      const cardsCta = document.querySelectorAll("webc-miam-recipe-card-cta");
      if (cardsCta === null) {
        mealzConsole('Not found CTA');
      }

      cardsCta.forEach((cardCta) => {
        cardCta.serves = event.detail.toString();
      });
      const pricing = document.querySelector("webc-miam-recipe-pricing");
      if (pricing === null || pricing === undefined) {
        mealzConsole('Not found Pricing');
      }
      pricing.serves = event.detail.toString();
    }

    initBasket() {
      const basketButton = document.querySelector('.mrtn-button-mealz-basket');
      if (!basketButton) {
        return;
      }
      basketButton.addEventListener('click', () => window.miam.basket.openPreview());

      this.refreshBasketNumber();
    }

    storeSubscribe() {
      mealzConsole('Init store subscribe');
      window.miam.events.storeLocatorOpened().subscribe({
                                                          next: opened => {
                                                            mealzConsole('store opened', opened);
                                                            if (opened) {
                                                              if (this.userLocation === null) {
                                                                this.initLocationUser()
                                                              } else {
                                                                this.sendPositionToMealz()
                                                              }
                                                            }
                                                          },
                                                        });
    }

    initLocationUser() {
      mealzConsole('Init location user');
      navigator.permissions.query({ name: "geolocation" }).then((result) => {
        if (result.state === "granted") {
          mealzConsole('prompt location user is granted');
          navigator.geolocation.getCurrentPosition((coords) => MealzManage.registerLocationAndSend(coords));
        } else if (result.state === "prompt") {
          mealzConsole('ask user location');
          navigator.geolocation.getCurrentPosition((coords) => MealzManage.registerLocationAndSend(coords));
        } else if (result.state === 'denied') {
          mealzConsole('denied location');
          MealzManage.registerLocationAndSend(undefined);
        }
        result.addEventListener("change", () => {
          mealzConsole('change permission of location');
          if (MealzManage.userLocation !== null && MealzManage.userLocation !== undefined) {
            MealzManage.sendPositionToMealz()
          }
        });
      });
    }

    registerLocationAndSend(value) {
      mealzConsole('Register location and send', value);
      this.userLocation = value;
      this.sendPositionToMealz();
    }

    sendPositionToMealz() {
      if (this.userLocation === null) {
        mealzConsole('userLocation are empty');
        return;
      }
      mealzConsole('location user send', this.userLocation);
      window.miam.user.setLocation(this.userLocation);
    }

                                                       return;
                                                     }

                                                     if (value > 0) {
                                                       spanNumber.style.display = 'block';
                                                       spanNumber.innerText = value;

                                                     } else {
                                                       spanNumber.style.display = 'none';
                                                     }

                                                   },
                                                   complete: () => {
                                                     console.log('recipeCount ok');
                                                   }
                                                 });
    }

    displayLog() {
      return this.logConstructor;
    }
  }

      }
    } else {
      setTimeout(setupMealz, 100);
    }
  }

  setupMealz();
